//Validacion javascript

(function() {
    var nombres=document.getElementById("nombre");
    var correos=document.getElementById("correo");
    var enviar=document.getElementById("send");
    enviar.addEventListener("click",verificacion);

    function verificacion(event) {

         if (nombres.value=="" || correos.value=="") {
              alert("Faltan datos");
              event.preventDefault();
         }
     }
}) ();

//Validacion jQuery

$(document).ready(function(){

    $("#form").submit(event => {
         if ($('#nombre').val().length==0 || $('#correo').val().length==0) {
              event.preventDefault();
              alert("Faltan datos");
          }
    });
});
